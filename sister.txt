<script>
  $(document).ready(function() {
    $('#submit').click(function(e) {
      e.preventDefault();
      var nama = $("#nama").val();
      var message = $('#message').val();
      var kategori = $('#kategori').val();
      var noTelp = $('#notelp').val();
      var file = $('#file')[0];
      var panjangNama = nama.replace(/\s/g, '').length;
      var panjangMessage = message.replace(/\s/g, '').length;
      var panjangNotelp = noTelp.length;
      var panjangFile = file.files.length

      var formData = new FormData($('#formkritik')[0]);

      if (panjangNama > 0 && panjangMessage > 0 && kategori != "" && panjangNotelp > 0 && panjangFile > 0) {
        $.ajax({
          type: "POST",
          url: "http://sister.uniqhba.ac.id/webservices/api/profildesa/kritikdansaran.php",
          data: formData,
          processData: false,
          contentType: false,
          success: function(response) {
            $('#forms').empty();
            $("#kritikgagal").remove()
            $('#forms').append(`
            <div class="alert alert-success">Kritik dan saran anda telah dikirimkan, Terima Kasih!</div>
            `)
            console.log(response);
          }
        });
      } else {
        $('#kritikgagal').remove();
        $('#forms').append(`
        <div id="kritikgagal" class="alert alert-danger">Mohon isi semua kolom</div>
        `)
      }
    })
  })
