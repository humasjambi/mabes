		$('#pengadaantable').on('click', '.tarik', function() {
        var nip = $(this).data('id');
		var query = $(this).attr('data-id');

        swal({
            title: "Yakin Edit Data?",
            text: "Ingin Menarik Data " + nip + " dengan URL " + query,
            icon: "warning",
            buttons: true,
            dangerMode: true,
        })
        .then((willEdit) => {
            if (willEdit) {
                $('#loading-modal').show();
                $.ajax({
                    type: 'GET',
                    url: query,
                    dataType: 'html',
                    success: function(response){
                        swal("Edit Data Berhasil!", { icon: "success" }).then(function(){
                            $('#loading-modal').hide();
                            location.reload();
                        });
                    },
                    error: function(xhr, status, error) {
                        $('#loading-modal').hide();
                        swal("Gagal melakukan edit data.", { icon: "error" });
                    }
                });
            } else {
                swal("Batal", {
                    text: "Edit data dibatalkan.",
                    icon: "warning",
                });
            }
        });
    });
